Module Type.Numeric

Poor man's type classes for numeric arguments.

type 'a integral =
  1. | Int : int integral
  2. | Int32 : int32 integral
  3. | Int64 : int64 integral

'a integral represents an integral type.

type 'a t =
  1. | Int : int t
  2. | Int32 : int32 t
  3. | Int64 : int64 t
  4. | Real : float t

'a t represents a numeric type.
